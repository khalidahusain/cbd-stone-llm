```python
import re
from dataclasses import dataclass
from typing import Optional

@dataclass
class SafeguardResult:
    safe: bool
    message: Optional[str] = None

class SafeguardService:
    """Pre-LLM injection defense and post-LLM output scanning.
    
    Implements SAFE-02 (injection defense) and SAFE-03 (post-generation scan).
    """

    # Patterns that indicate prompt injection attempts
    INJECTION_PATTERNS = [
        r"(?i)ignore\s+(all\s+)?previous\s+instructions",
        r"(?i)ignore\s+(all\s+)?prior\s+instructions",
        r"(?i)forget\s+(all\s+)?(previous|prior)\s+(context|instructions|rules)",
        r"(?i)disregard\s+(all\s+)?(previous|prior|above)",
        r"(?i)you\s+are\s+now\s+a",
        r"(?i)^system\s*:",
        r"(?i)\[system\]",
        r"(?i)\[INST\]",
        r"(?i)new\s+instructions?\s*:",
        r"(?i)override\s+(the\s+)?(system|extraction|prompt)",
        r"(?i)do\s+not\s+follow\s+(the\s+)?(above|previous|system)",
        r"(?i)act\s+as\s+(if\s+)?(you\s+are|a)",
        r"(?i)pretend\s+(you\s+are|to\s+be)",
        r"(?i)switch\s+(to|into)\s+\w+\s+mode",
    ]

    # Patterns that indicate the LLM generated clinical predictions
    PREDICTION_PATTERNS = [
        r"\d+\.?\d*\s*%",                              # "65%", "0.5%"
        r"(?i)probability\s+(of|is|was|=)",             # "probability of"
        r"(?i)likelihood\s+(of|is|was)",                # "likelihood of"
        r"(?i)(chance|risk)\s+(of|is)\s+\w+",           # "chance of", "risk is"
        r"(?i)(?:I|the model)\s+(?:predict|estimate|assess|calculate)", # "I predict", "the model estimates"
        r"(?i)(?:high|low|moderate|intermediate)\s+risk\s+(?:for|of)", # "high risk for"
        r"(?i)recommend\s+(?:ERCP|EUS|MRCP|IOC|surgery)", # clinical recommendations
        r"(?i)(?:should|must)\s+undergo\s+(?:ERCP|EUS|MRCP)", # directive recommendations
    ]

    def __init__(self):
        self._injection_re = [re.compile(p) for p in self.INJECTION_PATTERNS]
        self._prediction_re = [re.compile(p) for p in self.PREDICTION_PATTERNS]

    def check_input(self, user_input: str) -> SafeguardResult:
        """Check user input for prompt injection patterns before sending to LLM.
        
        Returns SafeguardResult(safe=True) if input is clean.
        Returns SafeguardResult(safe=False, message=...) if injection detected.
        """
        for pattern in self._injection_re:
            if pattern.search(user_input):
                return SafeguardResult(
                    safe=False,
                    message="Input rejected: your message contains patterns that could interfere with the extraction system. Please rephrase your clinical description.",
                )
        return SafeguardResult(safe=True)

    def check_output(self, llm_response: str) -> SafeguardResult:
        """Scan LLM output for probability values or clinical predictions.
        
        Returns SafeguardResult(safe=True) if output is clean.
        Returns SafeguardResult(safe=False, message=...) if prediction content detected.
        """
        for pattern in self._prediction_re:
            if pattern.search(llm_response):
                return SafeguardResult(
                    safe=False,
                    message="Response blocked: the extraction assistant attempted to generate clinical predictions. Only the validated ML model produces predictions.",
                )
        return SafeguardResult(safe=True)
```

```python
from openai import AsyncOpenAI
from typing import Optional

from backend.app.core.schema_loader import FeatureSchema
from backend.app.core.prompt_builder import PromptBuilder
from backend.app.schemas.extraction import ExtractionResult
from backend.app.services.safeguard import SafeguardService, SafeguardResult

class ExtractionError(Exception):
    """Raised when extraction fails due to safeguard or API issues."""
    def __init__(self, message: str, safeguard_triggered: bool = False):
        super().__init__(message)
        self.safeguard_triggered = safeguard_triggered

class ExtractionService:
    """Extracts clinical variables from free-text using OpenAI structured outputs.
    
    Uses GPT-4o-mini with Pydantic ExtractionResult as response_format.
    Integrates SafeguardService for pre-input and post-output checking.
    """

    MODEL = "gpt-4o-mini"
    TEMPERATURE = 0  # deterministic extraction
    MAX_TOKENS = 1000

    def __init__(
        self,
        schema: FeatureSchema,
        openai_client: AsyncOpenAI,
        safeguard: SafeguardService,
    ):
        self.schema = schema
        self.client = openai_client
        self.safeguard = safeguard
        self.system_prompt = PromptBuilder.build_system_prompt(schema)

    async def extract(self, user_input: str) -> ExtractionResult:
        """Extract clinical variables from clinician's free-text input.
        
        Args:
            user_input: Raw clinician text describing a patient case.
        
        Returns:
            ExtractionResult with extracted features (null for unmentioned fields).
        
        Raises:
            ExtractionError: If input fails safeguard check or LLM produces blocked content.
        """
        # Step 1: Pre-LLM input guard (SAFE-02)
        input_check = self.safeguard.check_input(user_input)
        if not input_check.safe:
            raise ExtractionError(input_check.message, safeguard_triggered=True)

        # Step 2: Build messages with data boundary (D-09)
        messages = [
            {"role": "system", "content": self.system_prompt},
            {"role": "user", "content": PromptBuilder.build_user_message(user_input)},
        ]

        # Step 3: Call OpenAI with structured output (D-01, SAFE-04)
        # SDK has built-in retry for 429/5xx (D-18). If still fails, let exception propagate.
        response = await self.client.beta.chat.completions.parse(
            model=self.MODEL,
            messages=messages,
            response_format=ExtractionResult,
            temperature=self.TEMPERATURE,
            max_tokens=self.MAX_TOKENS,
        )

        # Step 3b: Check for model refusal (D-19)
        if response.choices[0].message.refusal:
            raise ExtractionError(
                f"Model refused extraction: {response.choices[0].message.refusal}",
                safeguard_triggered=False,
            )

        result: ExtractionResult = response.choices[0].message.parsed

        # Step 4: Post-generation output scan (SAFE-03)
        # Check the raw response content for any probability patterns
        raw_content = response.choices[0].message.content or ""
        output_check = self.safeguard.check_output(raw_content)
        if not output_check.safe:
            raise ExtractionError(output_check.message, safeguard_triggered=True)

        return result
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User text -> SafeguardService -> OpenAI | Untrusted clinician input must pass injection check before reaching LLM |
| OpenAI response -> SafeguardService -> application | LLM output must pass prediction scan before reaching the user |
| ExtractionService -> InferenceService | These must NEVER be connected — architectural isolation enforced |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-0202-01 | Spoofing | User input | mitigate | SafeguardService.check_input() rejects injection patterns pre-LLM (SAFE-02) |
| T-0202-02 | Tampering | LLM output | mitigate | SafeguardService.check_output() blocks probability patterns post-LLM (SAFE-03) |
| T-0202-03 | Elevation of Privilege | LLM -> inference | mitigate | InferenceService never imported/referenced by ExtractionService; automated test verifies (SAFE-01) |
| T-0202-04 | Information Disclosure | API key in responses | accept | OpenAI client is server-side only; key never appears in HTTP responses |
</threat_model>
