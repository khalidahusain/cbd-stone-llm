```python
class FeatureSchema:
    features: list[FeatureDefinition]
    risk_tiers: list[RiskTier]
    cost_configs: dict[str, CostConfig]
    cholangitis_message: str
    version: str

    def get_model_columns(self) -> list[str]:
        return [f.model_column for f in self.features]

    def get_required_features(self) -> list[str]:
        return [f.name for f in self.features if f.required]

    def get_feature_by_name(self, name: str) -> Optional[FeatureDefinition]:
        for f in self.features:
            if f.name == name:
                return f
        return None

    def get_feature_by_column(self, column: str) -> Optional[FeatureDefinition]:
        for f in self.features:
            if f.model_column == column:
```

```python
from backend.app.core.schema_loader import FeatureSchema, FeatureDefinition
from backend.app.schemas.prediction import ValidationErrorDetail

class ValidationService:
    def __init__(self, schema: FeatureSchema):
        self.schema = schema

    def validate(self, features: dict[str, any]) -> list[ValidationErrorDetail]:
        """Validate feature values against YAML schema ranges and allowed values.
        
        Args:
            features: dict keyed by clinical feature name (e.g., "sex", "age", "ast")
        
        Returns:
            List of ValidationErrorDetail for any violations. Empty list = valid.
        """
        errors = []
        for name, value in features.items():
            if value is None:
                continue
            feature_def = self.schema.get_feature_by_name(name)
            if feature_def is None:
                errors.append(ValidationErrorDetail(
                    error="unknown_field",
                    field=name,
                    message=f"Unknown feature: {name}"
                ))
                continue
            
            if feature_def.type == "numeric" and feature_def.valid_range:
                if not isinstance(value, (int, float)):
                    errors.append(ValidationErrorDetail(
                        error="validation_error",
                        field=name,
                        message=f"{feature_def.name} must be a number"
                    ))
                elif value < feature_def.valid_range["min"] or value > feature_def.valid_range["max"]:
                    errors.append(ValidationErrorDetail(
                        error="validation_error",
                        field=name,
                        message=f"{feature_def.name} must be between {feature_def.valid_range['min']} and {feature_def.valid_range['max']} {feature_def.unit or ''}"
                    ))
            
            if feature_def.type == "categorical" and feature_def.allowed_values:
                if value not in feature_def.allowed_values:
                    errors.append(ValidationErrorDetail(
                        error="validation_error",
                        field=name,
                        message=f"{feature_def.name} must be one of: {', '.join(feature_def.allowed_values)}"
                    ))
            
            if feature_def.type == "boolean" and feature_def.encoding:
                allowed = list(feature_def.encoding.keys())
                if isinstance(value, str) and value not in allowed:
                    errors.append(ValidationErrorDetail(
                        error="validation_error",
                        field=name,
                        message=f"{feature_def.name} must be one of: {', '.join(allowed)}"
                    ))
        
        return errors
```

```python
import joblib
import pandas as pd
import numpy as np
from pathlib import Path
from typing import Union

from backend.app.core.schema_loader import FeatureSchema
from backend.app.schemas.prediction import (
    PredictionResult, CostEstimate, InsufficientInfoResult, BilirubinWarning
)

class InferenceService:
    def __init__(self, schema: FeatureSchema):
        self.schema = schema
        self.models: dict = {}

    def load_models(self, models_dir: str | Path) -> None:
        """Load all 6 pkl files from the models directory."""
        models_path = Path(models_dir)
        self.models = {
            "initial": joblib.load(models_path / "initial.pkl"),
            "imputer": joblib.load(models_path / "iterative_imputer.pkl"),
            "ercp": joblib.load(models_path / "model_predict_if_ercp.pkl"),
            "eus": joblib.load(models_path / "model_predict_if_eus.pkl"),
            "mrcp": joblib.load(models_path / "model_predict_if_mrcp.pkl"),
            "ioc": joblib.load(models_path / "model_predict_if_ioc.pkl"),
        }

    def predict(
        self, features: dict[str, any]
    ) -> Union[PredictionResult, InsufficientInfoResult]:
        """Run the full inference pipeline.
        
        Args:
            features: dict keyed by clinical feature name from YAML schema.
                     e.g., {"sex": "Male", "age": 55, "ast": 120, ...}
                     Missing optional fields can be omitted or set to None.
        
        Returns:
            PredictionResult on success, InsufficientInfoResult if required fields missing.
        """
        # Step 1: Check required fields (per D-03: sex and age)
        missing_required = []
        for feat in self.schema.features:
            if feat.required and (feat.name not in features or features[feat.name] is None):
                missing_required.append(feat.name)
        
        if missing_required:
            return InsufficientInfoResult(
                error="insufficient_information",
                missing_required=missing_required,
                message=f"Cannot run model: missing required fields: {', '.join(missing_required)}"
            )
        
        # Step 2: Build DataFrame with model column names
        # Use schema ordering for columns
        model_columns = self.schema.get_model_columns()
        row = {}
        imputed_fields = []
        
        for feat in self.schema.features:
            clinical_name = feat.name
            model_col = feat.model_column
            value = features.get(clinical_name)
            
            if value is None:
                row[model_col] = np.nan
                imputed_fields.append(clinical_name)
            else:
                # Apply encoding (YES/NO -> 1/0, Male/Female -> 1/0)
                if feat.encoding and isinstance(value, str):
                    row[model_col] = feat.encoding.get(value, value)
                else:
                    row[model_col] = value
        
        # Step 3: Create DataFrame and impute
        patient_df = pd.DataFrame([row], columns=model_columns)
        patient_imputed = self.models["imputer"].transform(patient_df)
        patient_imputed = pd.DataFrame(patient_imputed, columns=model_columns)
        
        # Step 4: Run initial model prediction
        prediction = np.round(
            self.models["initial"].predict_proba(patient_imputed)[0] * 100, 2
        )
        probability = float(prediction[1])  # index [1] = probability of stone
        
        # Step 5: Determine risk tier and intervention
        risk_tier = self.schema.get_risk_tier(probability)
        
        # Step 6: Calculate costs
        costs_dict = self.schema.calculate_costs(probability)
        cost_estimates = [
            CostEstimate(intervention=name, cost=cost)
            for name, cost in costs_dict.items()
        ]
        
        # Step 7: Check cholangitis fast-path (per D-06, MODL-07)
        cholangitis_value = features.get("clinical_cholangitis")
        cholangitis_flag = cholangitis_value in (True, "YES", 1)
        cholangitis_msg = self.schema.cholangitis_message if cholangitis_flag else None
        
        return PredictionResult(
            probability=probability,
            risk_tier=risk_tier.name,
            recommended_intervention=risk_tier.intervention,
            cost_estimates=cost_estimates,
            cholangitis_flag=cholangitis_flag,
            cholangitis_message=cholangitis_msg,
            imputed_fields=imputed_fields,
        )
```

| T-02-02 | Denial of Service | InferenceService.predict | accept | Single-user local deployment; no rate limiting needed at this phase. run_in_executor wrapping deferred to Plan 03 |
| T-02-03 | Spoofing | Feature dict keys | mitigate | ValidationService rejects unknown field names not in schema; InferenceService only reads fields defined in FeatureSchema |
| T-02-04 | Information Disclosure | Model internals | accept | pkl files are server-side only; predict() returns probability and tier, not model weights or feature importances |
</threat_model>
