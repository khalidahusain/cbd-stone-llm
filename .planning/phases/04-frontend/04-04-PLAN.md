```tsx
import { ChatProvider } from "./context/ChatContext";
import ChatPanel from "./components/ChatPanel";
import DashboardPanel from "./components/DashboardPanel";

function App() {
  return (
    <ChatProvider>
      <div className="flex h-screen bg-gray-50">
        {/* Header */}
        <div className="absolute top-0 left-0 right-0 h-10 bg-slate-800 flex items-center px-4 z-10">
          <h1 className="text-white text-sm font-semibold">CBD Stone LLM — Clinical Decision Support</h1>
        </div>
        
        {/* Main content below header */}
        <div className="flex w-full pt-10 h-screen">
          {/* Chat panel: left ~2/3 */}
          <div className="w-2/3 border-r border-gray-200">
            <ChatPanel />
          </div>
          
          {/* Dashboard panel: right ~1/3 */}
          <div className="w-1/3 bg-white overflow-y-auto">
            <DashboardPanel />
          </div>
        </div>
      </div>
    </ChatProvider>
  );
}

export default App;
```

The layout uses Tailwind flexbox:
- `h-screen` for full viewport height
- `w-2/3` / `w-1/3` for the panel split
- Minimal header (40px) to maximize clinical content area
- Dashboard panel has `overflow-y-auto` for smaller viewports (D-05)
